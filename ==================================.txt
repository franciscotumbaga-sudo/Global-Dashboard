function switchTool() {
    const selected = document.getElementById('masterSelector').value;
    document.querySelectorAll('.tool-section').forEach(sec => sec.classList.remove('active'));
    document.getElementById('tool_' + selected).classList.add('active');
    localStorage.setItem('activeTool_v2', selected);
}

function initGlobal() {
    // Load Tool
    const savedTool = localStorage.getItem('activeTool_v2');
    if (savedTool) {
        document.getElementById('masterSelector').value = savedTool;
        switchTool();
    }

    // Specific dropdown restoration for Wrap Code
    const savedKeywordHtml = localStorage.getItem('draft_w_key_html_v2');
    if (savedKeywordHtml) {
        document.getElementById('w_key').innerHTML = savedKeywordHtml;
    }

    // Restore InfoBox visibility for Wrap Code
    const savedDef = localStorage.getItem('draft_w_defText_v2');
    if (savedDef) {
        document.getElementById('w_defText').innerText = savedDef;
        document.getElementById('w_infoBox').style.display = 'block';
    }

    // Load AutoSave Drafts & attach listeners
    document.querySelectorAll('.auto-save').forEach(input => {
        const savedValue = localStorage.getItem('draft_' + input.id + '_v2');
        if (savedValue !== null) { input.value = savedValue; }
        
        input.addEventListener('input', () => {
            localStorage.setItem('draft_' + input.id + '_v2', input.value);
            triggerGen(input.id);
        });
        input.addEventListener('change', () => {
            localStorage.setItem('draft_' + input.id + '_v2', input.value);
            if(input.id === 'w_key') {
                localStorage.setItem('draft_w_key_html_v2', document.getElementById('w_key').innerHTML);
            }
            triggerGen(input.id);
        });
    });

    updateCommentUI();
    updateMacroUI();

    generateWrap(); displayHistory('w');
    generateComment(); displayHistory('c');
    generateRTS(); displayHistory('r');
    generateMacro();
}

function triggerGen(id) {
    if (id.startsWith('w_')) generateWrap();
    if (id.startsWith('c_')) generateComment();
    if (id.startsWith('r_')) generateRTS();
    if (id.startsWith('m_')) generateMacro();
}

function clearDrafts(prefix) {
    document.querySelectorAll('.auto-save').forEach(input => {
        if (input.id.startsWith(prefix + '_')) {
            localStorage.removeItem('draft_' + input.id + '_v2');
            if(input.tagName !== 'SELECT' || input.id === 'w_cat' || input.id === 'w_key') {
                input.value = ''; 
            }
        }
    });

    if(prefix === 'w') { 
        document.getElementById('w_status').value = 'Resolved'; 
        document.getElementById('w_issue').value = 'Customer reached out to '; 
        document.getElementById('w_key').innerHTML = '<option value="">-- Select Code First --</option>';
        document.getElementById('w_infoBox').style.display = 'none';
        localStorage.removeItem('draft_w_key_html_v2');
        localStorage.removeItem('draft_w_defText_v2');
    }
    if(prefix === 'c') { document.getElementById('c_type').value = 'tl_needed'; document.getElementById('c_comp').value = 'Refund'; document.getElementById('c_tltog').value = 'yes'; updateCommentUI(); }
    triggerGen(prefix + '_');
}

async function copyToolOutput(sourceId, btnId) {
    const area = document.getElementById(sourceId);
    const btn = document.getElementById(btnId);
    if (!area.value.trim()) return;

    area.select();
    try { await navigator.clipboard.writeText(area.value); } 
    catch (err) { document.execCommand('copy'); }

    const originalText = btn.innerText;
    btn.innerText = "✓ COPIED!";
    btn.style.background = "#fff";
    btn.style.color = "#000";
    setTimeout(() => { btn.innerText = originalText; btn.style.background = ""; btn.style.color = ""; }, 1000);
}

document.getElementById('w_search').addEventListener('input', function() {
    const term = this.value.toLowerCase().trim();
    const resList = document.getElementById('w_results');
    resList.innerHTML = '';
    if(term.length < 1) { resList.style.display = 'none'; return; }

    const matches = wrapDB.filter(item => 
        item.title.toLowerCase().includes(term) || 
        item.code.toLowerCase().includes(term) ||
        item.keywords.some(k => k.toLowerCase().includes(term))
    );

    if (matches.length > 0) {
        resList.style.display = 'block';
        matches.forEach(item => {
            const div = document.createElement('div');
            div.className = 'result-item';
            div.innerHTML = `<strong>${item.title}</strong><span class="result-snippet">${item.code}</span>`;
            div.onclick = () => selectWrap(item);
            resList.appendChild(div);
        });
    } else { resList.style.display = 'none'; }
});

function selectWrap(item) {
    const cat = document.getElementById('w_cat');
    const keySel = document.getElementById('w_key');
    
    cat.value = item.code;
    localStorage.setItem('draft_w_cat_v2', cat.value); 

    document.getElementById('w_infoBox').style.display = 'block';
    document.getElementById('w_defText').innerText = item.definition;
    localStorage.setItem('draft_w_defText_v2', item.definition);
    
    keySel.innerHTML = '<option value="">-- Select Specific Keyword --</option>';
    item.keywords.forEach(k => {
        const opt = document.createElement('option');
        opt.value = k; opt.innerText = k; keySel.appendChild(opt);
    });
    
    localStorage.setItem('draft_w_key_html_v2', keySel.innerHTML);
    localStorage.setItem('draft_w_key_v2', '');

    document.getElementById('w_results').style.display = 'none';
    document.getElementById('w_search').value = '';
    generateWrap();
}

    res.value = res.value ? res.value + " " + text : text;
    localStorage.setItem('draft_w_res_v2', res.value);
    generateWrap();
}

function generateWrap() {
    const key = document.getElementById('w_key').value || "[No Keyword Selected]";
    const iss = document.getElementById('w_issue').value;
    const res = document.getElementById('w_res').value;
    const stat = document.getElementById('w_status').value;
    const oth = document.getElementById('w_other').value;

    let resStr = res ? `[${stat}] ${res}` : `[${stat}]`;
    
    // Notice how 'Wrap Code: ...' is completely gone now!
    let final = `Keyword: ${key}\nIssue: ${iss}\nResolution: ${resStr}`;
    
    if (oth.trim() !== "") { 
        final += `\nOther: ${oth}`; 
    }
    
    document.getElementById('w_out').value = final;
}

function updateCommentUI() {
    const val = document.getElementById('c_type').value;
    document.getElementById('c_box_email').classList.toggle('hidden', val !== 'cc_error');
    document.getElementById('c_box_recipe').classList.toggle('hidden', val !== 'quality_missing');
    document.getElementById('c_box_tl').classList.toggle('hidden', val === 'no_tl');
    document.getElementById('c_box_tltog').classList.toggle('hidden', val !== 'quality_missing');
    generateComment();
}

function generateComment() {
    const val = document.getElementById('c_type').value;
    const comm = document.getElementById('c_comm').value.trim();
    const amt = document.getElementById('c_amt').value.trim();
    const cType = document.getElementById('c_comp').value;
    const tl = document.getElementById('c_tl').value.trim();
    const rec = document.getElementById('c_rec').value.trim();
    const email = document.getElementById('c_email').value.trim();
    const tlNeed = document.getElementById('c_tltog').value;

    const compPart = (amt && cType !== 'N/A') ? `${cType} $${amt}` : '';
    const tlPart = (tl && (val !== 'no_tl' && (val !== 'quality_missing' || tlNeed === 'yes'))) ? `TL: ${tl}` : '';
    const recPart = (val === 'quality_missing' && rec) ? `R#${rec}` : '';
    const emailPart = (val === 'cc_error' && email) ? `CC: ${email}` : '';

    document.getElementById('c_out').value = [emailPart, comm, recPart, tlPart, compPart].filter(Boolean).join(" | ");
}

function getHistoryKey(prefix) {
    if(prefix === 'w') return 'hist_wrap_v2';
    if(prefix === 'c') return 'hist_comm_v2';
    if(prefix === 'r') return 'hist_rts_v2';
}

function saveHistory(prefix) {
    const out = document.getElementById(prefix + '_out').value;
    if (!out.trim()) return;
    
    const key = getHistoryKey(prefix);
    const history = JSON.parse(localStorage.getItem(key) || '[]');
    history.unshift({ id: Date.now(), time: new Date().toLocaleString(), text: out });
    if(history.length > 30) history.pop();
    
    localStorage.setItem(key, JSON.stringify(history));
    displayHistory(prefix);
    clearDrafts(prefix); 
}

function displayHistory(prefix) {
    const history = JSON.parse(localStorage.getItem(getHistoryKey(prefix)) || '[]');
    const log = document.getElementById(prefix + '_log');
    log.innerHTML = history.length === 0 ? '<p style="font-size:11px; color:#888; text-align:center;">Log empty.</p>' : '';
    
    history.forEach(item => {
        const div = document.createElement('div');
        div.className = 'history-item';
        div.innerHTML = `<span class="del-btn" onclick="delHist('${prefix}', ${item.id})">×</span><div class="timestamp">${item.time}</div><div>${item.text}</div>`;
        log.appendChild(div);
    });
}

function delHist(prefix, id) {
    const key = getHistoryKey(prefix);
    let history = JSON.parse(localStorage.getItem(key) || '[]');
    history = history.filter(i => i.id !== id);
    localStorage.setItem(key, JSON.stringify(history));
    displayHistory(prefix);
}

function clearHistory(prefix) {
    if(confirm("Clear this entire log?")) {
        localStorage.removeItem(getHistoryKey(prefix));
        displayHistory(prefix);
    }
}

function exportHistory(prefix, name) {
    const history = JSON.parse(localStorage.getItem(getHistoryKey(prefix)) || '[]');
    if (!history.length) return alert("History is empty!");
    let content = `${name.toUpperCase()} LOG\n======================\n\n`;
    history.forEach(item => content += `[${item.time}]\n${item.text}\n---\n\n`);
    const a = document.createElement('a');
    a.href = URL.createObjectURL(new Blob([content], { type: 'text/plain' }));
    a.download = `${name}_${new Date().toISOString().slice(0,10)}.txt`;
    a.click();
}

document.addEventListener('click', (e) => {
    const sw = document.getElementById('w_search');
    const rl = document.getElementById('w_results');
    if (sw && rl && !sw.contains(e.target) && !rl.contains(e.target)) { rl.style.display = 'none'; }
});
